32 object HandleRejection {
33   def newHandler(): RejectionHandler =
34     RejectionHandler
35       .newBuilder()
36       .handle {
37         case MissingQueryParamRejection(msg) =>
38           complete((StatusCodes.BadRequest, msg))
39       }
40       .handle {
41         case ValidationRejection(msg, _) =>
42           complete((StatusCodes.BadRequest, msg))
43       }
44       .handleNotFound {
45         complete(StatusCodes.NotFound)
46       }
47       .result()
48       .withFallback(RejectionHandler.default)
49       .mapRejectionResponse {
50         case res @ HttpResponse(status, headers, ent: HttpEntity.Strict, _) =>
51           // since all Akka default rejection responses are Strict this will handle all rejections
52           val message = ent.data.utf8String.replaceAll("\"", """\"""")
53 
54           // we copy the response in order to keep all headers and status code, wrapping the message as hand rolled JSON
55           // you could the entity using your favourite marshalling library (e.g. spray json or anything else)
56           res.withHeadersAndEntity(
57             headers = headers,
58             HttpEntity(
59               ContentTypes.`application/json`,
60               s"""{"code": ${status.intValue}, "type": "${status.reason}", "message": "$message"}""")
61           )
62 
63         case x => x // pass through all other types of responses
64       }
65 }
