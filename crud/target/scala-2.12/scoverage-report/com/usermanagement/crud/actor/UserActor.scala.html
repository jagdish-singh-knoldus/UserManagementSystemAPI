28 class UserActor(userDAO: UserDAO)(implicit futureAwaitDuration: FiniteDuration)
29     extends FailurePropatingActor
30     with ActorLogging
31     with LazyLogging
32     with Constants {
33 
34   import UserActor._
35 
36   //noinspection ScalaStyle
37   override def receive: Receive = {
38     case m @ GetAllUserDetails =>
39       val res = getAllUserDetails
40       res.pipeTo(sender())
41     case m @ ValidateUserId(userId: String) =>
42       val res = validateUserIdForUser(userId)
43       res.pipeTo(sender())
44     case m @ ValidateUserType(userId: String, userType: String) =>
45       val res = validateUserTpe(userId, userType)
46       res.pipeTo(sender())
47     case m @ InsertUserDetails(userDetails: UserDetails) =>
48       val res = insertUserDetails(userDetails)
49       res.pipeTo(sender())
50     case m @ UpdateUsernameForAdmin(userId: String,
51                                     username: String,
52                                     userType: String) =>
53       val res = updateAdminUsername(userId, username, userType)
54       res.pipeTo(sender())
55     case m @ DeleteUser(userId: String, userType: String) =>
56       val res = deleteUser(userId, userType)
57       res.pipeTo(sender())
58   }
59 
60   /**
61     * function to get user details
62     * @return Future[APIDataResponse]
63     */
64   private def getAllUserDetails(): Future[APIDataResponse] = {
65     userDAO.getAllUsers.map {
66       case Nil =>
67         logger.info("No data Found")
68         NoDataFound()
69       case data =>
70         GetUserDetailsSuccess(data)
71     }
72   }
73 
74   /**
75     * function to validate user id
76     * @param userId
77     * @return Future[Valid]
78     */
79   private def validateUserIdForUser(userId: String): Future[Valid] =
80     userDAO
81       .validateUserId(userId)
82       .map(count => Valid(count > 0))
83 
84   /**
85     * function to validate user type
86     * @param userId
87     * @param userType
88     * @return Future[Valid]
89     */
90   private def validateUserTpe(userId: String, userType: String): Future[Valid] =
91     userDAO
92       .validateUserType(userId, userType)
93       .map(count => Valid(count > 0))
94 
95   /**
96     * function to  insert user details
97     * @param userDetails
98     * @return Future[APIDataResponse]
99     */
100   private def insertUserDetails(
101       userDetails: UserDetails): Future[APIDataResponse] = {
102     userDAO.createUser(userDetails).map {
103       case 1 =>
104         CommandSuccess("User created successfully.", userDetails.userId)
105       case 0 =>
106         CommandFailure("User creation failed!")
107     }
108   }
109 
110   /**
111     * function t update admin username
112     * @param userId
113     * @param username
114     * @param userType
115     * @return Future[APIDataResponse]
116     */
117   private def updateAdminUsername(userId: String,
118                                   username: String,
119                                   userType: String): Future[APIDataResponse] = {
120     userDAO.updateUsername(userId, username, userType).map {
121       case 1 =>
122         CommandSuccess("Admin username updated successfully.", Some(userId))
123       case 0 =>
124         CommandFailure("Admin username updation failed!")
125     }
126   }
127 
128   /**
129     * function to delete user
130     * @param userId
131     * @param userType
132     * @return Future[APIDataResponse]
133     */
134   private def deleteUser(userId: String,
135                          userType: String): Future[APIDataResponse] = {
136     userDAO.deleteUser(userId, userType).map {
137       case 1 =>
138         CommandSuccess("Customer deleted successfully.", Some(userId))
139       case 0 =>
140         CommandFailure("Customer deletion failed!")
141     }
142   }
143 
144 }
